  ret = ''
  i=0
  while A[i] == B[i]:
    ret += A[i]
    i += 1
  return ret

def common( A, B ):
  longest = ''
  for a_pos in len(A):
    for b_pos in len(B):
      m = longest_match(A[a_pos:], B[b_pos:])
